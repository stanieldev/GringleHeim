class Player {
public:
    int posX;
    int posY;
    std::vector<int> charisma;
    Player(std::vector<double> posVector, std::vector<int> charmArray) {
        posX = posVector.at(0);
        posY = posVector.at(1);
        charisma = charmArray;
    };
};




class Render {
public:
    // Extrema Render Distance
    const double X_MAX = 16.0; // 16.000 Blocks Radius Max
    const double Y_MAX = 9.0;  // 9.0000 Blocks Radius
    const double X_MIN = 1.0;  // 1.0000 Block Radius
    const double Y_MIN = 9/16; // 0.5625 Block Radius

    // Scale Factor
    double scaleFactor;

    // Initial Position
    int initX;
    int initY;

    // Render Distance
    int radiusX;
    int radiusY;

    Render(std::vector<int> res, double scale) {
        scaleFactor = scale;
        if (scaleFactor > res[0] / (32.0 * X_MAX) || 
            scaleFactor < res[1] / (32.0 * Y_MIN)
        ) { scaleFactor = 2; }

        initX = res[0] / 2;
        initY = res[1] / 2;

        radiusX = res[0] / (32 * scaleFactor) + 1;
        radiusY = res[1] / (32 * scaleFactor) + 1;

        if (radiusX - 1 > X_MAX)  { radiusX = X_MAX + 1; }
        else if (radiusX < X_MIN) { radiusX = X_MIN + 1; }
        if (radiusY - 1 > Y_MAX)  { radiusY = Y_MAX + 1; }
        else if (radiusY < Y_MIN) { radiusY = Y_MIN + 1; }
    }
};
